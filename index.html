import random def hangman(): words = ['apple', 'banana', 'cat', 'dog', 'happy', 'sun', 'tree', 'book', 'smile', 'star'] word = random.choice(words) guessed = ['_'] * len(word) attempts = 6 guessed_letters = set() print("Welcome to Hangman!") print("Guess the word: " + " ".join(guessed)) while attempts > 0 and '_' in guessed: guess = input("Enter a letter: ").lower() if not guess.isalpha() or len(guess) != 1: print("Please enter a single letter.") continue if guess in guessed_letters: print(f"You already guessed '{guess}'. Try another letter.") continue guessed_letters.add(guess) if guess in word: for i, letter in enumerate(word): if letter == guess: guessed[i] = guess print("Good guess: " + " ".join(guessed)) else: attempts -= 1 print(f"Wrong guess. Attempts left: {attempts}") print(" ".join(guessed)) if '_' not in guessed: print(f"Congratulations! You guessed the word '{word}'.") else: print(f"Game over! The word was '{word}'.") if __name__ == "__main__": hangman()
